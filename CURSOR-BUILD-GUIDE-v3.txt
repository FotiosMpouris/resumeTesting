Luminous Motes Canvas (The Atmosphere That Makes It Special)
This is the single biggest “wow” element. Warm golden/cream dust motes rising slowly like in a sunbeam.Place a single <canvas id="motes" class="motes-canvas"> with position: fixed; z-index: 1; pointer-events: none; behind everything.Full optimized particle code (put in script.js):
js
class Mote {
  constructor(canvas) {
    this.canvas = canvas;
    this.reset();
  }
  reset() {
    this.x = Math.random() * this.canvas.width;
    this.y = Math.random() * this.canvas.height;
    this.radius = Math.random() * 2.2 + 0.8;
    this.opacity = Math.random() * 0.12 + 0.06;
    this.vy = -0.12 - Math.random() * 0.18; // upward drift
    this.vx = (Math.random() - 0.5) * 0.15;
    this.hue = 38 + Math.random() * 8; // warm gold/cream
    this.targetOpacity = this.opacity;
    this.flareTimer = 0;
  }
  update(scrollY = 0) {
    this.y += this.vy;
    this.x += this.vx + (scrollY * 0.008); // subtle parallax
    if (this.y < -10) this.y = this.canvas.height + 10;
    if (this.x < 0 || this.x > this.canvas.width) this.x = Math.random() * this.canvas.width;


    // Occasional flare (catches the light)
    this.flareTimer++;
    if (this.flareTimer > 280 && Math.random() < 0.008) {
      this.targetOpacity = 0.38;
      this.radius = this.radius * 1.4;
      this.flareTimer = 0;
    }
    this.opacity = this.opacity * 0.96 + this.targetOpacity * 0.04;
    if (this.opacity < this.targetOpacity + 0.01) this.targetOpacity = this.opacity * 0.7 + 0.05;
  }
  draw(ctx) {
    ctx.save();
    ctx.fillStyle = `hsla(${this.hue}, 72%, 78%, ${this.opacity})`;
    ctx.beginPath();
    ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
    ctx.fill();
    ctx.restore();
  }
}


let motes = [];
let ctx, canvas;


function initMotes() {
  canvas = document.getElementById('motes');
  ctx = canvas.getContext('2d', { alpha: true });
  resizeMotes();
  for (let i = 0; i < 48; i++) motes.push(new Mote(canvas));
  window.addEventListener('resize', resizeMotes);
  animateMotes();
}


function resizeMotes() {
  canvas.width = window.innerWidth;
  canvas.height = window.innerHeight;
}


let lastScrollY = 0;
function animateMotes() {
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  const scrollY = window.scrollY;
  const parallax = scrollY - lastScrollY;
  lastScrollY = scrollY;


  motes.forEach(mote => {
    mote.update(parallax);
    mote.draw(ctx);
  });
  requestAnimationFrame(animateMotes);
}


// Call initMotes() after DOMContentLoaded
This creates exactly the “dust caught in a sunbeam” feeling. It’s lightweight and beautiful.
